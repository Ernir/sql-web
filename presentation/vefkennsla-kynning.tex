\begin{center}
    \begin{minipage}{0.9\textwidth}
        \begin{figure}

            \label{fig:markdown-internal-link}
            \begin{syntaxenv}{markdown-internal-link}
                `[' `[' <identifier>
                \begin{stack}
                    `|' <label>\\

            \end{syntaxenv}
            \begin{syntaxenv}{markdown-footnote}
                `[' `\textasciicircum' <note-identifier> `]' `[' <contents> `]'

\end{center}
\end{frame}
